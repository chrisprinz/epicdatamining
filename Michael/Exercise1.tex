\item{10,000 documents

$IDF \\ = log_2(10,000,000/10,000) \\ = 10$}
\end{enumerate}
\item
\begin{enumerate}
\item{once

$TF_{wd} = 1/15$

$IDF_w = log_2(10,000,000/320) \\ = 15$

$TF.IDF = 1/15 * 15 \\ = 1$}

$TF_{wd} = 5/15 \\ = 1/3$

$TF.IDF = 1/3 * 15 \\ = 5$}
